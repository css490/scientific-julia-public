function cost(X, y, theta)
    m = length(y)  
    C = 0 #cost
    for i=1:m
        C += (1/m)*(-y[i]*log(sigmoid(X[i,:]*theta)) - (1 - y[i])*log(1-sigmoid(X[i,:]*theta)))
    end
    return C
end

function costGrad(X, y, theta)
    m = length(y)  
    delta = zeros(size(theta)) #partial derivatives with respect to each theta
    for i=1:m
        delta += (1/m)*(sigmoid(X[i,:]*theta) - y[i])[1]*(X[i,:])'
    end
    return delta
end

function gradientDescent(X, y, theta, alpha, iter)
    m = length(y)
    Jhist = zeros(iter,1)
    for k=1:iter
        delta = zeros(size(theta)) #partial derivatives with respect to each theta
        for i=1:m
            delta += (1/m)*(sigmoid(X[i,:]*theta) - y[i])[1]*(X[i,:])'
        end
        theta = theta - (alpha/m)*delta
        Jhist[k] = cost(X, y, theta)[1]
    end
    return theta, Jhist
end
