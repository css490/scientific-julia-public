function featureNorm(X)
    X_norm = X
    mu = zeros(1, size(X,2))
    sigma = zeros(1, size(X,2))
    
    for j = 1:size(X,2)
        mu[j] = mean(X[:,j])
        sigma[j] = std(X[:,j])
    end
    
    for i = 1:size(X,1)
        X_norm[i,:] = (X[i,:] - mu)./sigma
    end
    
    return X_norm
end

function computeCost(X, y, theta)
    m = length(y)  
    J=0
    for i=1:m
        J += (0.5/m)*(X[i,:]*theta - y[i])^2
    end
    return J
end

function gradientDescent(X, y, theta, alpha, iter)
    m = length(y)
    n = size(X,2)
    Jhist = zeros(iter,1)
    for i=1:iter
        delta=zeros(n,1)
        for j=1:m
            delta +=(X[j,:]*theta - y[j])[1]*(X[j,:])'
        end
        theta = theta - (alpha/m)*delta
        Jhist[i] = computeCost(X, y, theta)[1]
    end
    return theta, Jhist
end
